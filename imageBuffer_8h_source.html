13 // ImageElementTraits
14 //
15 // For each class of value, tells ImageBuffer how to store it: how much
16 // memory an element takes up, and what kind of pointer to assume.
17 //
18 // By default, this is a pointer to the type and its sizeof(.). However,
19 // in some cases this won't work, e.g. for a half or something which doesn't
20 // have a particular type. In that case, void can be used as the type, but the
21 // length can be correct; specialisation is necessary

98 template <typename MemType>
99 ImageBuffer<MemType>::ImageBuffer(cl::Context context,
100  cl_mem_flags flags,
101  size_t width, size_t height,
102  size_t padding, size_t alignment,
103  size_t numSlices)
104  : width_(width),
105  height_(height),
106  padding_(padding),
107  stride_(width + 2*padding),
108  numSlices_(numSlices)
109 {
110  // Stride might need extending to respect alignment
111  size_t overshoot = stride_ % alignment;
112  if (overshoot != 0)
113  stride_ += alignment - overshoot;
114 
115  // Pad the height to meet the alignment as well, in case
116  // anyone ever tried to access that region of memory
117  size_t fullHeight = height_ + 2*padding_;
118  overshoot = fullHeight % alignment;
119  if (overshoot)
120  fullHeight += alignment - overshoot;
121 
122  // Record the location of the upper left pixel, linear index
123  // into the buffer
124  start_ = stride_ * padding_ + padding_;
125  pitch_ = fullHeight * stride_;
126 
127  buffer_ = cl::Buffer {
128  context, flags,
129  numSlices_ * pitch_ * ImageElementTraits<MemType>::size
130  };
131 }

155  const MemType* input,
156  const std::vector<cl::Event> events,
157  cl::Event* done) const
158 {
159  const size_t numElements = buffer_.getInfo<CL_MEM_SIZE>()
160  / ImageElementTraits<MemType>::size;
161 
162  std::vector<MemType> bufferContents(numElements);
163 
164  // Copy into the output buffer row by row
165  for (int s = 0; s < numSlices_; ++s) {
166 
167  auto writePos = bufferContents.begin() + start_
168  + s * pitch_;;
169  for (int n = 0; n < height_; ++n, writePos += stride_,
170  input += width_)
171  std::copy(input, input + width_, writePos);
172 
173  }
